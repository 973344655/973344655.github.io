```
import java.io.InputStream;
import java.nio.file.Files;
import java.nio.file.Paths;
import java.util.Properties;

public class ProperGet {
    public static final String defaultPath = "configs.properties";
    private static Logger log = LoggerFactory.getLogger(ProperGet.class);

    public static Properties getProper(String propertyPath) {
        Properties properties = new Properties();
        try {
            properties.load(Files.newInputStream(Paths.get(propertyPath)));
        } catch (Exception e) {
            log.error("读取配置文件失败{}",e.getMessage());
        }
        return properties;
    }

    public static Properties getProper() {
        Properties properties = new Properties();
        try {
            InputStream in = ProperGet.class.getClassLoader().getResourceAsStream(defaultPath);
            properties.load(in);
        } catch (Exception e) {
            log.error("读取配置文件失败{}",e.getMessage());
        }
        return properties;
    }
}
```
